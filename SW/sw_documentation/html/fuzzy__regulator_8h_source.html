88  float regulator(float _actual_point, float _error, float _derror = 0, float _ierror = 0) {
89  float result = 0;
90  // The evaluation is made between the intervals, that's why the for gets only to NUM_FUZZY_REGULATOR_NODES-1 so it never reaches to evaluate nodes[NUM_FUZZY_REGULATOR_NODES+1], which will cause a segfault
91  for (uint8_t i = 0; i < NUM_FUZZY_REGULATOR_NODES-1; i++) {
92  if (_actual_point > nodes[i].getAplicationPoint() &&
93  _actual_point <= nodes[i+1].getAplicationPoint()) {
94  // A triangular membership function is applied:
95  // Vertex A is the smaller, vertex B is the bigger, the point to evaluate is between both
96  float vertexA = nodes[i].regulator(_error, _derror, _ierror);
97  float vertexB = nodes[i+1].regulator(_error, _derror, _ierror);
98  result = (_actual_point - vertexA) / (vertexB - vertexA);
99  i++;
100  } else {
101  nodes[i].regulator(_error, _derror, _ierror);
102  }
103  }
